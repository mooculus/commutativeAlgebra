Recall that a local ring is a ring that is Noetherian with a unique maximal ideal.\index{local ring}

\begin{definition}[Third notion of dimension]\index{sM@$s(M)$}
  Let $(A,\m)$ be a local ring and $M$ a finitely generated
  $A$-module. Define
  \[
  s(M) := \inf\left\{d:
  \begin{minipage}{35ex}
    there exists $x_1,\dots, x_d\in \m$ such that $\l(M/(x_1,\dots,x_d)M)<\infty$
  \end{minipage} \right\}.
  \]
\end{definition}

\begin{definition}\index{system of parameters}
  Let $R$ be a Noetherian ring and $M$ be a finitely generated
  $R$-module. If $s(M) = n$, then any sequence $x_1,\dots, x_n$ such
  that
  \[
  \l(M/(x_1,\dots,x_n)M)<\infty\index{finite length}
  \]
  is called a \dfn{system of parameters} for $M$. 
\end{definition}
